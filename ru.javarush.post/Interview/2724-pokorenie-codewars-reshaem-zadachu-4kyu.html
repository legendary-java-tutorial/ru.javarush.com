public class SudokuSolutionValidator {
    public static void main(String[] args) {
        int[][] matrix = {
                        {5, 3, 4,  6, 7, 8,  9, 1, 2},
                        {6, 7, 2,  1, 9, 5,  3, 4, 8},
                        {1, 9, 8,  3, 4, 2,  5, 6, 7},
                        {8, 5, 9,  7, 6, 1,  4, 2, 3},
                        {4, 2, 6,  8, 5, 3,  7, 9, 1},
                        {7, 1, 3,  9, 2, 4,  8, 5, 6},
                        {9, 6, 1,  5, 3, 7,  2, 8, 4},
                        {2, 8, 7,  4, 1, 9,  6, 3, 5},
                        {3, 4, 5,  2, 8, 6,  1, 7, 9}};
        System.out.println(validSolution(matrix));
    }

    public static boolean validSolution(int[][] matrix) {
//  –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞—à—É –º–∞—Ç—Ä–∏—Ü—É –Ω–∞ –Ω–∞–ª–∏—á–∏–µ 0
        for (int[] array : matrix) {
            for (int element : array) {
                if (element == 0) return false;
            }
        }
        if ((checkVerticalAndHorizontal(matrix))&&(check3v3squares(matrix)))
            return true;
        else return false;
    }

//   –í –º–µ—Ç–æ–¥–µ checkVerticalAndHorizontal() –ø—Ä–æ–≤–µ—Ä–∏–º –Ω–∞ –ø—Ä–∞–≤–∏–ª—å–Ω–æ—Å—Ç—å –∑–∞–ø–æ–ª–Ω–µ–Ω–∏—è
//   –Ω–∞—à–∏ —Å—Ç—Ä–æ–∫–∏ –∏ —Å—Ç–æ–ª–±—Ü—ã
    private static boolean checkVerticalAndHorizontal (int[][] matrix) {

//  4 –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ, –ø—Ä–µ–¥–Ω–∞–∑–Ω–∞—á–µ–Ω–∏–µ –∫–æ—Ç–æ—Ä—ã—Ö –≤—ã —É–∑–Ω–∞–µ—Ç –Ω–∏–∂–µ)
        int vertical = 0;
        int horizontal = 0;
        int horizontalCount = 0;
        int verticalCount = 0;

        for (int i = 0; i < 9; i++) {
            for (int j = 1; j < 10; j++) {

                for (int k = 0; k < 9; k++) {
                    if (matrix[i][k] == j) {
                        horizontalCount++;
//  –í –¥–∞–Ω–Ω–æ–º —Å–ª—É—á–∞–µ, –º—ã –ø—Ä–æ–≤–µ—Ä—è–ª–∏ –Ω–∞–ª–∏—á–∏–µ –ø–æ–≤—Ç–æ—Ä–æ–≤ –≤ —Å—Ç—Ä–æ–∫–∞—Ö,
//  –ï—Å–ª–∏ –≤—ã —É–±–µ—Ä–µ—Ç–µ break (–≤—ã—Ö–æ–¥ –∏–∑ —Ü–∏–∫–ª–∞ "k"), –∏ —Ä–∞—Å–∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–π—Ç–µ
//  –∫–æ–¥ –Ω–∏–∂–µ, –≤—ã —É–≤–∏–¥–∏—Ç–µ –∫–∞–∫–æ–≥–æ —á–∏—Å–ª–∞ –Ω–µ —Ö–≤–∞—Ç–∞–µ—Ç, –∞ –∫–∞–∫–æ–π —ç–ª–µ–º–µ–Ω—Ç –¥—É–±–ª–∏—Ä—É–µ—Ç—Å—è –¥–≤–∞–∂–¥—ã –∏ –±–æ–ª–µ–µ
//  System.out.println("matrix[i][k] = " + matrix[i][k] + "–∑–Ω–∞—á–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–∞ = " + j+"_______"+ "–∏–Ω–¥–µ–∫—Å —ç–ª–µ–º–µ–Ω—Ç–∞ –≤ —Å—Ç—Ä–æ–∫–µ "+k);
                        break;
                    }
                }


                for (int k = 0; k < 9; k++) {
//  –ê —Ç—É—Ç –ø—Ä–æ–≤–µ—Ä—è–µ–º —Å—Ç–æ–ª–±—Ü—ã. –ö–∞–∫ –≤—ã –ø–æ–Ω–∏–º–∞–µ—Ç–µ, –Ω–∞–ª–∏—á–∏–µ
//  –æ—à–∏–±–∫–∏ –≤ —Å—Ç—Ä–æ–∫–µ —Å–≤–∏–¥–µ—Ç–µ–ª—å—Å—Ç–≤—É–µ—Ç –æ–± –æ—à–∏–±–∫–µ –≤ —Å—Ç–æ–ª–±—Ü–µ :D
                    if (matrix[k][i] == j) {
                        verticalCount++;
//              —ç–ª–µ–º–µ–Ω—Ç —Å–æ–≤–ø–∞–ª +, —ç–ª–µ–º–µ–Ω—Ç –Ω–µ —Å–æ–≤–ø–∞–ª 0.
//              –µ—â–µ —Ä–∞–∑ –ø–æ–≤—Ç–æ—Ä—é—Å—å, break –Ω—É–∂–µ–Ω –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –Ω–µ –ª–æ–≤–∏—Ç—å –ø–æ–≤—Ç–æ—Ä—ã.
                        break;
                    }
                }
//  –î–∞–ª–µ–µ –∑–∞–∫–æ–º–º–µ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–∞ –º–µ—Ç–∫–∞, –∫–æ—Ç–æ—Ä–∞—è —Å–ª—É–∂–∏—Ç –Ω–∞–≤–∏–≥–∞—Ç–æ—Ä–æ–º –ø–æ —Ü–∏–∫–ª–∞–º
//               System.out.println("______________________");

            }
            if (horizontalCount == 9) horizontal++;
            if (verticalCount == 9) vertical++;
            horizontalCount = 0;
            verticalCount = 0;
//  –î–≤–µ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏ if –≤—ã—à–µ –∫–∞–∫ —Ä–∞–∑ –∏ —Ä–µ–∞–ª–∏–∑—É—é—Ç –º–æ—é –æ—Å–Ω–æ–≤–Ω—É—é –∏–¥–µ—é –ø—Ä–æ–≤–µ—Ä–∫–∏ —Å—Ç–æ–ª–±—Ü–æ–≤ –∏ —Å—Ç—Ä–æ–∫,
//  –∞ –∏–º–µ–Ω–Ω–æ –ø—Ä–∏ –ø–µ—Ä–µ–±–æ—Ä–µ –≤—Å–µ—Ö –∑–Ω–∞—á–µ–Ω–∏–π (–æ—Ç 1 –¥–æ 9) –≤ —Ü–∏–∫–ª–∞—Ö k –º—ã –ø—Ä–æ–≤–µ—Ä—è–º —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç –ª–∏
//  —Å—Ç—Ä–æ–∫–∞ –∏ —Å—Ç–æ–ª–±–µ—Ü –∑–∞–¥–∞–Ω–Ω–æ–º—É —É—Å–ª–æ–≤–∏—é –∏–ª–∏ –Ω–µ—Ç.
        }
//  –µ—Å–ª–∏ –Ω–∞—à–∏ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ vertical –∏ horizontal —Ä–∞–≤–Ω—ã 9 –∫–∞–∂–¥–∞—è,
//  –∑–Ω–∞—á–∏—Ç —Å–æ —Å—Ç–æ–ª–±—Ü–∞–º–∏ –∏ —Å—Ç—Ä–æ–∫–∞–º–∏ –ø—Ä–æ–±–ª–µ–º –Ω–µ—Ç.
        if ((vertical == 9)&&(horizontal==9)) return true;
        else return false;
    }

//  –í –º–µ—Ç–æ–¥–µ check3v3squares() –ø–µ—Ä–µ—Ö–æ–¥–∏–º –∫ —Å–ª–∞–¥–∫–æ–º—É, –ø—Ä–æ–≤–µ—Ä–∫–∞ –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ 3–Ω–∞3
    private static boolean check3v3squares(int[][] matrix) {

        int transitionDown = 0;
        int transitionRight = 0;
        int lastCount = 0;
//  –ü–æ –∫–≤–∞–¥—Ä–∞—Ç–∞–º –±—É–¥–µ–º –¥–≤–∏–≥–∞—Ç—å—Å—è –≤–Ω–∏–∑, –¥–æ–π–¥—è –¥–æ –Ω–∏–∑–∞ –Ω–∞ —à–∞–≥ –≤–ª–µ–≤–æ.
//  –ü–æ –¥–æ—Å—Ç–∏–∂–µ–Ω–∏—é –ø—Ä–∞–≤–æ–≥–æ –Ω–∏–∂–Ω–µ–≥–æ –∫–≤–∞–¥—Ä–∞—Ç–∞ –≤—ã—Ö–æ–¥–∏–º –∏–∑ —Ü–∏–∫–ª–∞. ( –≤ —Å—É–º–º–µ 9 –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ (3–Ω–∞3), —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ
//  –≤–Ω–µ—à–Ω–∏–π –∫—Ä—É–≥ –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å 9 –∏—Ç–µ—Ä–∞—Ü–∏–π)

        for ( int x = 0; x<9; x++) {
            if(transitionDown == 9) {
                transitionRight +=3;
                transitionDown = 0;
            }
//  –í —Ü–∏–∫–ª–∞—Ö –Ω–∏–∂–µ —è —Ä–µ–∞–ª–∏–∑–æ–≤–∞–ª–∏ –¥–≤–∏–∂–µ–Ω–∏–µ –ø–æ –∫–≤–∞–¥—Ä–∞—Ç–∞–º 3—Ö3 –∏ –∏—Ö –ø—Ä–æ–≤–µ—Ä–∫—É.
//  –ü–µ—Ä–µ–º–µ–Ω–Ω–∞—è key –Ω—É–∂–Ω–∞ –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø–æ–≤—Ç–æ—Ä–Ω–æ–π –ø—Ä–æ–≤–µ—Ä–∫–∏ –æ–¥–Ω–æ–≥–æ
//  –∏ —Ç–æ–≥–æ –∂–µ —á–∏—Å–ª–∞ (—Ç–æ–µ—Å—Ç—å –ø—Ä–∏ –Ω–∞–ª–∏—á–∏–∏ 2 –æ–¥–∏–Ω–∞–∫–æ–≤—ã—Ö —á–∏—Å–µ–ª, –º—ã —É–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—á–µ—Ç—á–∏–∫ –Ω–∞ 1, –∞ –Ω–µ 2)
            for (int k = 1; k < 10; k++) {
                int key = 0;
                for (int i = 0; i < 3; i++) {
                        for (int j = 0; j < 3; j++) {
                        if ((k == matrix[i+transitionDown][j+transitionRight])&&(key!=k)) {
                            lastCount++;
                            key = k;
                        }
                    }
                }
            }
            transitionDown += 3;
        }
//  –°—á–µ—Ç—á–∏–∫ lastCount –¥–æ–ª–∂–µ–Ω –≤–µ—Ä–Ω—É—Ç—å —á–∏—Å–ª–æ 81, –µ—Å–ª–∏ –æ–Ω–æ –º–µ–Ω—å—à–µ, —Ç–æ –Ω–µ –æ–±–æ—à–ª–æ—Å—å –±–µ–∑ –ø–æ–≤—Ç–æ—Ä–æ–≤
        if(lastCount == 81) return true;
        else return false;
    }
}
